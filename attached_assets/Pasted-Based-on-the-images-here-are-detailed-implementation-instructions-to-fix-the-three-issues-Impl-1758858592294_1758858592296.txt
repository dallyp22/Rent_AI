### 1. **Auto-Adjust Sliders Based on Optimization Goal Selection** ðŸŽ¯

When users select different optimization goals, the sliders should automatically adjust to optimal values for that strategy.

#### Implementation:
```typescript
// components/OptimizationTab/OptimizationControls.tsx

import { useState, useEffect } from 'react';
import { RadioGroup, RadioGroupItem } from '@/components/ui/radio-group';
import { Slider } from '@/components/ui/slider';

// Preset configurations for each optimization goal
const OPTIMIZATION_PRESETS = {
  'Maximize Revenue': {
    targetOccupancy: 85,  // Lower occupancy acceptable for higher rates
    riskTolerance: 'High' // Willing to take risks for revenue
  },
  'Maximize Occupancy': {
    targetOccupancy: 98,  // Target near-full occupancy
    riskTolerance: 'Low'  // Conservative approach to ensure occupancy
  },
  'Balanced': {
    targetOccupancy: 92,  // Balance between revenue and occupancy
    riskTolerance: 'Medium' // Moderate risk approach
  },
  'Custom': {
    targetOccupancy: null, // Keep current values
    riskTolerance: null    // Keep current values
  }
};

const OptimizationControls = ({ onParametersChange }) => {
  const [optimizationGoal, setOptimizationGoal] = useState('Maximize Revenue');
  const [targetOccupancy, setTargetOccupancy] = useState(95);
  const [riskTolerance, setRiskTolerance] = useState('Medium');
  
  // Auto-adjust sliders when optimization goal changes
  useEffect(() => {
    const preset = OPTIMIZATION_PRESETS[optimizationGoal];
    
    if (preset && optimizationGoal !== 'Custom') {
      // Animate the slider transitions
      if (preset.targetOccupancy !== null) {
        animateSliderValue(
          targetOccupancy,
          preset.targetOccupancy,
          (value) => setTargetOccupancy(value)
        );
      }
      
      if (preset.riskTolerance !== null) {
        setRiskTolerance(preset.riskTolerance);
      }
      
      // Notify parent component
      onParametersChange({
        optimizationGoal,
        targetOccupancy: preset.targetOccupancy || targetOccupancy,
        riskTolerance: preset.riskTolerance || riskTolerance
      });
    }
  }, [optimizationGoal]);
  
  // Smooth animation for slider transitions
  const animateSliderValue = (from, to, setter) => {
    const duration = 500; // ms
    const steps = 20;
    const stepDuration = duration / steps;
    const stepSize = (to - from) / steps;
    
    let currentStep = 0;
    const interval = setInterval(() => {
      currentStep++;
      const newValue = from + (stepSize * currentStep);
      setter(Math.round(newValue));
      
      if (currentStep >= steps) {
        clearInterval(interval);
        setter(to); // Ensure exact final value
      }
    }, stepDuration);
  };
  
  const getRiskToleranceValue = (tolerance) => {
    const map = { 'Low': 33, 'Medium': 66, 'High': 100 };
    return map[tolerance];
  };
  
  const getRiskToleranceLabel = (value) => {
    if (value <= 33) return 'Low';
    if (value <= 66) return 'Medium';
    return 'High';
  };
  
  return (
    <div className="space-y-8">
      {/* Optimization Goal Selection */}
      <div>
        <h3 className="text-sm font-medium mb-4">Optimization Goal</h3>
        <RadioGroup
          value={optimizationGoal}
          onValueChange={setOptimizationGoal}
        >
          <div className="space-y-2">
            <label className="flex items-center space-x-2 cursor-pointer">
              <RadioGroupItem value="Maximize Revenue" />
              <span>Maximize Revenue</span>
            </label>
            <label className="flex items-center space-x-2 cursor-pointer">
              <RadioGroupItem value="Maximize Occupancy" />
              <span>Maximize Occupancy</span>
            </label>
            <label className="flex items-center space-x-2 cursor-pointer">
              <RadioGroupItem value="Balanced" />
              <span>Balanced</span>
            </label>
            <label className="flex items-center space-x-2 cursor-pointer">
              <RadioGroupItem value="Custom" />
              <span>Custom</span>
            </label>
          </div>
        </RadioGroup>
      </div>
      
      {/* Target Occupancy Slider */}
      <div>
        <h3 className="text-sm font-medium mb-2">Target Occupancy</h3>
        <div className="px-2">
          <Slider
            value={[targetOccupancy]}
            onValueChange={([value]) => setTargetOccupancy(value)}
            min={85}
            max={100}
            step={1}
            disabled={optimizationGoal !== 'Custom'}
            className={optimizationGoal !== 'Custom' ? 'opacity-75' : ''}
          />
          <div className="flex justify-between text-xs text-gray-500 mt-2">
            <span>85%</span>
            <span className="font-medium text-blue-600">{targetOccupancy}%</span>
            <span>100%</span>
          </div>
        </div>
      </div>
      
      {/* Risk Tolerance Slider */}
      <div>
        <h3 className="text-sm font-medium mb-2">Risk Tolerance</h3>
        <div className="px-2">
          <Slider
            value={[getRiskToleranceValue(riskTolerance)]}
            onValueChange={([value]) => {
              setRiskTolerance(getRiskToleranceLabel(value));
            }}
            min={0}
            max={100}
            step={33}
            disabled={optimizationGoal !== 'Custom'}
            className={optimizationGoal !== 'Custom' ? 'opacity-75' : ''}
          />
          <div className="flex justify-between text-xs text-gray-500 mt-2">
            <span>Low</span>
            <span className="font-medium text-blue-600">{riskTolerance}</span>
            <span>High</span>
          </div>
        </div>
      </div>
      
      {/* Visual indicator when not in custom mode */}
      {optimizationGoal !== 'Custom' && (
        <div className="bg-blue-50 border border-blue-200 rounded-lg p-3">
          <p className="text-sm text-blue-700">
            Sliders are automatically configured for "{optimizationGoal}" strategy. 
            Select "Custom" to manually adjust values.
          </p>
        </div>
      )}
    </div>
  );
};
```

```typescript
// utils/formatters.ts

/**
 * Format currency values to 2 decimal places
 * Handles floating point precision issues
 */
export const formatCurrency = (value: number | string): string => {
  if (value === null || value === undefined) return '$0.00';
  
  // Convert to number if string
  const numValue = typeof value === 'string' ? parseFloat(value) : value;
  
  // Handle NaN
  if (isNaN(numValue)) return '$0.00';
  
  // Use toFixed(2) and ensure proper rounding
  return `$${numValue.toFixed(2)}`;
};

/**
 * Format large currency values with commas
 */
export const formatLargeCurrency = (value: number | string): string => {
  if (value === null || value === undefined) return '$0.00';
  
  const numValue = typeof value === 'string' ? parseFloat(value) : value;
  
  if (isNaN(numValue)) return '$0.00';
  
  return new Intl.NumberFormat('en-US', {
    style: 'currency',
    currency: 'USD',
    minimumFractionDigits: 2,
    maximumFractionDigits: 2
  }).format(numValue);
};
```
